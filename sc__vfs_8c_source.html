193 int _sys_chmod(char *path, mode_t mode)
194 {
195  inode_t *inode = vfs_find_inode(path);
196  if (!inode) {
197  syscall_errno = ENOENT;
198  return -1;
199  }
200  if (get_perm_class(inode->uid, inode->gid) != PERM_CLASS_OWNER) {
201  syscall_errno = EPERM;
202  return -1;
203  }
204  if (get_effective_uid() != 0)
205  mode &= 0777;
206  inode->mode &= ~07777;
207  inode->mode |= mode & 07777;
208  inode->ctime = system_time;
209  return 0;
210 }

233 int _sys_chown(char *path, uid_t owner, gid_t group)
234 {
235  inode_t *inode = vfs_find_inode(path);
236  if (!inode) {
237  syscall_errno = ENOENT;
238  return -1;
239  }
240  if (get_perm_class(inode->uid, inode->gid) != PERM_CLASS_OWNER) {
241  syscall_errno = EPERM;
242  return -1;
243  }
244  if (get_perm_class(-1, group) > PERM_CLASS_GROUP)
245  group = 65535;
246  if (group != 65535)
247  inode->gid = group;
248  if (owner != 65535)
249  inode->uid = owner;
250  inode->ctime = system_time;
251  return 0;
