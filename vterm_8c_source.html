32 void vterm_init(vterm_t *vterm, guint flags, int rows, int cols)
33 {
34  int row;
35  /* record dimensions */
36  vterm->rows=rows;
37  vterm->cols=cols;
38 
39  vterm->cells = heapmm_alloc(rows * sizeof(vterm_cell_t *));
40  for (row = 0; row < rows; row++){
41  vterm->cells[row] = heapmm_alloc(cols * sizeof(vterm_cell_t));
42 
43  }
44  vterm->colors = 0x07;
45  vterm->curattr = 0x07;
46  // initialize all cells with defaults
47  vterm_erase(vterm);
48 
49  // initialization of other public fields
50  vterm->crow=0;
51  vterm->ccol=0;
52  vterm->state = 0;
53  // default active colors

56  vterm->scroll_min=0;
57  vterm->scroll_max=rows-1;

61 
62  vterm->flags=flags;
63 
64  if(flags & VTERM_FLAG_VT100)
65  vterm->write=vterm_write_vt100;
