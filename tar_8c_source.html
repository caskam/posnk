108  int status;
109  tar_header_t *header = (tar_header_t *) (tar_data + (uintptr_t)*pos);

111  if (strlen(header->name) == 0){
112  return 1;
113  }
114  if (strcmp(header->magic, TMAGIC) != 0) {
115  earlycon_printf("WARNING: invalid tar record for file %s magic:\"%s\" read:%i pos:%i\n",header->name,header->magic,*pos);
116  if (strlen(header->name) <= 1) {
117  return EIO;
118  }
119 
120  }
121  //debugcon_printf("untar %s %c \n",header->name, header->typeflag);
122  switch(header->typeflag){
123  default:
124  case AREGTYPE:
125  case REGTYPE:
126  case CONTTYPE:
127  status = vfs_mknod(header->name, tar_get_mode(header), 0);
128  if (status) {
129  debugcon_printf("WARNING: error creating file while extracting %s, errno: %i\n",header->name, status);
130  return status;
131  }
132  file = vfs_find_inode(header->name);
133  if (!file) {
134  debugcon_printf("WARNING: error opening file while extracting %s, errno: %i\n",header->name, status);
135  return ENOSPC;
136  }
137  vfs_truncate(file, (off_t) tar_num_dec(header->size,12));
138  file->mtime = file->atime = file->ctime = (time_t) tar_num_dec(header->mtime, 12);
139  if (file->size != 0) {
140  status = vfs_write(file, 0, (void*)(tar_data + (uintptr_t)*pos), file->size, &wd_count, 0);
141  (*pos) += tar_round_up(file->size,512);
142  if (status) {

144  return status;
145  }

148  return ENOSPC;

151  break;

164  }
165  break;

196 int tar_read_record(inode_t *tar_file, off_t *pos)
197 {
198  void *buf;
199  inode_t *file;
200  size_t rd_count;
201  size_t wd_count;
202  off_t wr_count = 0;
203  off_t wr_size = 0;
204  off_t block_size = 512*2*512;
205  off_t e_block_size = block_size;
206  int status;
207  tar_header_t *header = heapmm_alloc(sizeof(tar_header_t));
208  if (!header) {
209  return ENOMEM;
210  }
211  status = vfs_read(tar_file, *pos, header, sizeof(tar_header_t), &rd_count, 0);
212  if (status) {
213  heapmm_free(header, sizeof(tar_header_t));
214  earlycon_printf("WARNING: error read file while extracting %s, errno: %i\n",tar_file->name,status);
215  return status;
216  }
217  (*pos) += rd_count;
218  if (rd_count != sizeof(tar_header_t)) {
219  heapmm_free(header, sizeof(tar_header_t));
220  earlycon_printf("WARNING: size mismatch read file while extracting %s read:%i pos:%i, errno: %i\n",tar_file->name,status,rd_count,*pos);
221  return EIO;
222  }
223  if (strlen(header->name) == 0){
224  heapmm_free(header, sizeof(tar_header_t));
225  return 1;
226  }
227  if (strcmp(header->magic, TMAGIC) != 0) {
228  earlycon_printf("WARNING: invalid tar record for file %s magic:\"%s\" read:%i pos:%i\n",header->name,header->magic,rd_count,*pos);
229  if (strlen(header->name) <= 1) {

231  return EIO;
232  }
233 
234  }
235  switch(header->typeflag){
236  default:
237  case AREGTYPE:
238  case REGTYPE:
239  case CONTTYPE:
240  status = vfs_mknod(header->name, tar_get_mode(header), 0);
241  if (status) {
242  earlycon_printf("WARNING: error creating file while extracting %s, errno: %i\n",header->name, status);
243  heapmm_free(header, sizeof(tar_header_t));
244  return status;

253  if (file->size != 0) {
254  buf = heapmm_alloc(block_size);
255  if (!buf) {
256  heapmm_free(header, sizeof(tar_header_t));
257  return ENOSPC;
258  }
259  for (wr_count = 0; wr_count < file->size; wr_count += e_block_size) {
260  if ((file->size - wr_count) < block_size)
261  e_block_size = 512;
262  status = vfs_read(tar_file, *pos, buf, e_block_size, &rd_count, 0);
263  if (status) {
264  heapmm_free(header, sizeof(tar_header_t));
265  heapmm_free(buf, block_size);

267  return status;
268  }
269  (*pos) += rd_count;
270  if (rd_count != (size_t) e_block_size) {
271  earlycon_printf("WARNING: hit EOF while extracting %s\n",header->name);
272  heapmm_free(header, sizeof(tar_header_t));
273  heapmm_free(buf, block_size);
274  return EIO;
275  }
276  wr_size = file->size - wr_count;
277  if (wr_size > e_block_size)
278  wr_size = e_block_size;
279  status = vfs_write(file, wr_count, buf, wr_size, &wd_count, 0);
280  if (status) {

290  return ENOSPC;
291  }
292  }
293  heapmm_free(buf, block_size);
294  }
295  break;
296  case SYMTYPE://TODO: Implement proper symlinks
297  case LNKTYPE:
298  status = vfs_link(header->linkname, header->name);
299  if (status) {
300  heapmm_free(header, sizeof(tar_header_t));
301  return status;
302  }

304  case CHRTYPE:
