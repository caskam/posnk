156 void tty_input_char(dev_t device, char c)
157 {
158  aoff_t a;
159  tty_info_t *tty = tty_get(device);
160  if (tty->termios.c_lflag & ICANON) { //TODO: Flow control
161  if ((tty->termios.c_iflag & INLCR) && (c == '\n'))
162  c = '\r';
163  else if ((tty->termios.c_iflag & ICRNL) && (c == '\r'))
164  c = '\n';
165  if ((tty->termios.c_iflag & IGNCR) && (c == '\r'))
166  return;
167  if ((tty->termios.c_iflag & ISTRIP))
168  c &= 0x7F;
169  if ((c == tty->termios.c_cc[VERASE]) && (tty->termios.c_lflag & ECHOE)) {
170  if ((tty->line_buffer_pos)) {
171  tty->write_out(device, '\b');
172  tty->write_out(device, ' ');
173  tty->write_out(device, '\b');
174  }
175  } else if ((c == tty->termios.c_cc[VKILL]) && (tty->termios.c_lflag & ECHOK)) {
176  for (a = 0; a < tty->line_buffer_pos; a++) {
177  tty->write_out(device, '\b');
178  tty->write_out(device, ' ');
179  tty->write_out(device, '\b');
180  }
181  } else if ((tty->termios.c_lflag & ECHO) ||
182  ((c == '\n') && (tty->termios.c_lflag & ECHONL)))
183  tty->write_out(device, c);
184 
185  if ((c == '\n') || (c == tty->termios.c_cc[VEOL])) {
186  tty_buf_line_char(tty, c);
187  tty_flush_line_buffer(tty);
188  } else if (c == tty->termios.c_cc[VERASE]) {
189  if (tty->line_buffer_pos)
190  tty->line_buffer_pos--;
191  } else if (c == tty->termios.c_cc[VKILL]) {
192  tty->line_buffer_pos = 0;
193  } else if (tty->termios.c_lflag & ISIG) {
194  if (c == tty->termios.c_cc[VQUIT]) {
195  process_signal_pgroup(tty->fg_pgid, SIGQUIT);
196  } else if (c == tty->termios.c_cc[VINTR]) {
197  process_signal_pgroup(tty->fg_pgid, SIGINT);
198  } else if (c == tty->termios.c_cc[VSUSP]) {
199  process_signal_pgroup(tty->fg_pgid, SIGTSTP);
200  } else
201  tty_buf_line_char(tty, c);
202  } else {
203  tty_buf_line_char(tty, c);
204  }
205  } else if (tty->termios.c_lflag & ECHO) {
206  tty->write_out(device, c);
207  tty_buf_out_char(tty, c);
208  } else {
209  tty_buf_out_char(tty, c);
210  }
211 }
