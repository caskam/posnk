630 off_t _sys_lseek(int fd, off_t offset, int whence)
631 {
632  int _before_begin = 0;
633  aoff_t old_offset;
634  stream_ptr_t *ptr;
635 
636  /* Get the stream ptr for fd */
637  ptr = stream_get_ptr(fd);
638 
639  /* Check whether the fd exists */
640  if (!ptr) {
641  syscall_errno = EBADF;
642  return -1;
643  }
644 
645  /* Check whether the stream refers to a file */
646  if ((ptr->info->type != STREAM_TYPE_FILE) || (S_ISFIFO(ptr->info->inode->mode))) {
647  syscall_errno = ESPIPE;
648  return (off_t) -1;
649  }
650 
651  /* Acquire a lock on the stream */
652  semaphore_down(ptr->info->lock);
653 
654  /* Store the old offset so we can recover from errors */
655  old_offset = ptr->info->offset;
656 
657  /* Handle the different whence modes */
658  switch (whence) {
659  case SEEK_SET:
660  /* Just set the offset */
661  ptr->info->offset = (aoff_t) offset;
662  break;
663 
664  case SEEK_CUR:
665  /* Offset relative to current offset */
666 
667  /* Check whether offset is positive */
668  if (offset >= 0) {
669  /* It is, just add it to the offset */
670  ptr->info->offset += (aoff_t) offset;
671  } else {
672  /* This hack is necessary because file offset
673  * is unsigned and 2^32-1 is a valid offset */
674 
675  /* It is negative, check for negative result */
676  if (((aoff_t) -offset) > ptr->info->offset)
677  _before_begin = 1;
678 
679  /* Subtract it from the offset */
680  ptr->info->offset -= (aoff_t) -offset;
681  }
682 
683  break;
684  case SEEK_END:
685  /* Offset relative to EOF */
686 
687  /* Check whether offset is positive */
688  if (offset >= 0) {
689  /* It is, just add it to the size */
690  ptr->info->offset = ptr->info->inode->size + (aoff_t) offset;
691  } else {
692  /* This hack is necessary because file offset
693  * is unsigned and 2^32-1 is a valid offset */
694 
695  /* It is negative, check for negative result */
696  if (((aoff_t) -offset) > ptr->info->inode->size)
697  _before_begin = 1;
698 
699  /* Subtract it from the size */
700  ptr->info->offset = ptr->info->inode->size - (aoff_t) -offset;
701  }
702  break;
703 
704  default:
705  /* Invalid whence value */
706  /* Set before_begin to trigger an undo */
707  _before_begin = 1;
708  break;
709  }
710 
711  /* Check for errors */
712  if (_before_begin) {
713  /* Undo seek */
714  ptr->info->offset = old_offset;
715 
716  /* Release the lock on the stream */
717  semaphore_up(ptr->info->lock);
718 
719  /* Report error */
720  syscall_errno = EINVAL;
721  return (off_t) -1;
722  }
723 
724  /* Release the lock on the stream */
725  semaphore_up(ptr->info->lock);
726 
727  /* Return the new offset */
728  return (off_t) ptr->info->offset;
729 }
