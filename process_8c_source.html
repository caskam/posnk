78 void process_signal_handler_default(int signal)
79 {
80  //debugcon_printf("sigrecv: %i\n", scheduler_current_task->pid);
81  scheduler_current_task->last_signal = signal;
82  switch (signal_default_actions[signal]) {
83  case SIGNAL_ACTION_ABORT:
84  case SIGNAL_ACTION_TERM:
85  scheduler_current_task->term_cause = PROCESS_TERM_SIGNAL;
86  scheduler_current_task->state = PROCESS_KILLED;
87  //earlycon_printf("killed: %i\n", scheduler_current_task->pid);
88  break;
89  case SIGNAL_ACTION_CONT:
90  scheduler_current_task->state = PROCESS_RUNNING;
91  break;
92  case SIGNAL_ACTION_STOP:
93  if (scheduler_current_task->state != PROCESS_WAITING)
94  scheduler_current_task->waiting_on = NULL;
95  scheduler_current_task->state = PROCESS_WAITING;
96  break;
97  case SIGNAL_ACTION_IGNORE:
98  default:
99  return;
100  }

161 void process_handle_signals()
162 {
163 
164  uint32_t bit_counter;
165  uint32_t sig_active = scheduler_current_task->waiting_signal_bitmap & ~scheduler_current_task->signal_mask_bitmap;
166  //debugcon_printf("sigactive: %i\n", sig_active);
167  if (sig_active != 0) {
168  debugcon_printf("sighandle: %i\n", scheduler_current_task->pid);
169  if (scheduler_current_task->state != PROCESS_INTERRUPTED) {
170  for (bit_counter = 0; bit_counter < 32; bit_counter++) {
171  if (sig_active & (1 << bit_counter)) {
172  scheduler_current_task->waiting_signal_bitmap &= ~ (1 << bit_counter);
173  debugcon_printf("handling signal: %i\n", sig_active);
174  if (scheduler_current_task->signal_handler_table[bit_counter] == NULL)
175  process_signal_handler_default(bit_counter);
176  else if (scheduler_current_task->signal_handler_table[bit_counter] == 1)
177  continue;
178  else
179  scheduler_invoke_signal_handler(bit_counter);
180  if (scheduler_current_task->state == PROCESS_KILLED) {
181  debugcon_printf("killedby: %i\n", sig_active);
182  process_child_event(scheduler_current_task, PROCESS_CHILD_KILLED);
183  procvmm_clear_mmaps();
