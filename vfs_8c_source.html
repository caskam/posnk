1760 {

1762  dir_cache_t *newc;
1763  inode_t * parent = dirc->inode;
1764  char * separator;
1765  char * path_element;
1766  char * remaining_path = path;
1767  char * end_of_path;
1768  size_t element_size;
1769  dirent_t *dirent;
1770  int element_count = 0;
1771 
1772  /* Check for NULL or empty path */
1773  if ((path == NULL) || ((*path) == '\0'))
1774  return NULL;
1775 
1776  /* Allocate buffer for path element */
1777  path_element = (char *) heapmm_alloc(CONFIG_FILE_MAX_NAME_LENGTH);
1778 
1779  /* Aqquire a pointer to the terminator */
1780  end_of_path = strchr(remaining_path, 0);
1781 
1782  /* Iterate over the path */
1783  for (;;) {
1784  /* Find the next / in the path */
1785  separator = strchrnul(remaining_path, (int) '/');
1786 
1787  /* Calculate the size of this path element */
1788  element_size = ((uintptr_t) separator) - ((uintptr_t) remaining_path);
1789 
1790  /* If the first element has size 0, this path has a leading */
1791  /* / so it is to be considered an absolute path */
1792  if ((element_count == 0) && (element_size == 0)){ // First character is /
1793  /* Path is absolute */
1794 
1795  /* Update current element */
1796  dirc = scheduler_current_task->root_directory;
1797  parent = dirc->inode;
1798 
1799  } else if (element_size == 0) {
1800  /* Element size is 0 but not first element */
1801 
1802  /* Check whether this is the last element */
1803  if ((separator + 2) >= end_of_path) {
1804  /* If so, we have detected a trailing slash */
1805  /* and reached the end of the path */
1806 
1807  /* Bump the reference counters and return */
1808  dirc->usage_count++;
1809  dirc->inode->usage_count++;
1810 
1811  /* Clean up */
1812  heapmm_free(path_element, CONFIG_FILE_MAX_NAME_LENGTH);
1813 
1814  return dirc;
1815  }
1816 
1817  } else if ((element_size == 2) && !strncmp(remaining_path, "..", 2)) {
1818  /* Element is ".." */
1819 
1820  /* Set next current path element to the parent of
1821  * the current element */
1822  newc = dirc->parent;
1823 
1824  /* If the new current element differs from the current
1825  * element, free the old current path element */
1826  if (newc != dirc) {
1827 
1828  /* Current element is not in use elsewhere? */
1829  if ((!dirc->usage_count))
1830  /* Free it */
1831  heapmm_free(dirc, sizeof(dir_cache_t));
1832 
1833  /* Update current element */
1834  parent = newc->inode;
1835  dirc = newc;
1836  }
1837  } else if ((element_size == 1) && !strncmp(remaining_path, ".", 1)) {
1838  /* Element is . */
1839  /* Ignore this */
1840  } else {
1841  /* Element is a normal path element */
1842 
1843  /* Isolate it */
1844  strncpy(path_element, remaining_path, element_size);
1845  path_element[element_size] = '\0';
1846 
1847  //TODO: Check for search permission here or in vfs_find_dirent
1848 
1849  /* Find the directory entry for it */
1850  dirent = vfs_find_dirent(parent, path_element);
1851 
1852  /* Check if it exists */
1853  if (dirent == NULL) {
1854 
1855  /* If not, clean up and return */
1856  heapmm_free(path_element, CONFIG_FILE_MAX_NAME_LENGTH);
1857 
1858  /* Check if this is the last path element */
1859  if ((separator + 1) >= end_of_path) {
1860  /* We have already determined the
1861  * parent so even though the entry does
1862  * not exist, return its parent */
1863 
1864  /* Bump the reference counters and return */
1865  dirc->usage_count++;
1866  dirc->inode->usage_count++;
1867 
1868  return dirc;
1869  } else if (dirc && dirc->usage_count == 0){
1870  /* Free the current dirc if it is not referenced
1871  * somewhere else */
1872 
1873  /* Release its reference on its parent */
1874  vfs_dir_cache_release(dirc->parent);
1875 
1876  /* Free its memory */
1877  heapmm_free(dirc, sizeof(dir_cache_t));
1878  }
1879 
1880  /* An element in the path is nonexistent*/
1881 
1882  /* Return NULL */
1883  return NULL;
1884  }
1885 
1886  /* Bump reference counters on old path element as it
1887  * will be referenced by the new dirc entry */
1888  dirc->inode->usage_count++;
1889  dirc->usage_count++;
1890 
1891  /* Create new dirc entry for this element */
1892  newc = heapmm_alloc(sizeof(dir_cache_t));
