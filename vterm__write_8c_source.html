44 void vterm_write_rxvt(vterm_t *vterm,guint32 keycode)
45 {
46  gchar *buffer=NULL;
47 
48  switch(keycode)
49  {
50  case '\n': buffer="\r"; break;
51  case KEY_UP: buffer="\033[A"; break;
52  case KEY_DOWN: buffer="\033[B"; break;
53  case KEY_RIGHT: buffer="\033[C"; break;
54  case KEY_LEFT: buffer="\033[D"; break;
55  case KEY_BACKSPACE: buffer="\b"; break;
56  case KEY_IC: buffer="\033[2~"; break;
57  case KEY_DC: buffer="\033[3~"; break;
58  case KEY_HOME: buffer="\033[7~"; break;
59  case KEY_END: buffer="\033[8~"; break;
60  case KEY_PPAGE: buffer="\033[5~"; break;
61  case KEY_NPAGE: buffer="\033[6~"; break;
62  case KEY_SUSPEND: buffer="\x1A"; break; // ctrl-z
63  case KEY_F(1): buffer="\033[11~"; break;
64  case KEY_F(2): buffer="\033[12~"; break;
65  case KEY_F(3): buffer="\033[13~"; break;
66  case KEY_F(4): buffer="\033[14~"; break;
67  case KEY_F(5): buffer="\033[15~"; break;
68  case KEY_F(6): buffer="\033[17~"; break;
69  case KEY_F(7): buffer="\033[18~"; break;
70  case KEY_F(8): buffer="\033[19~"; break;
71  case KEY_F(9): buffer="\033[20~"; break;
72  case KEY_F(10): buffer="\033[21~"; break;
73  case KEY_F(11): buffer="\033[23~"; break;
74  case KEY_F(12): buffer="\033[24~"; break;
75  }
76 
77  if(buffer==NULL)
78  vterm_write_tty(vterm, &keycode,sizeof(char));
79  else
80  vterm_write_tty(vterm, buffer,strlen(buffer));
81 
82  return;
83 }
