278 void ata_setup_lba_transfer(ata_device_t *device, int drive, ata_lba_t lba, uint16_t count)
279 {
280 #ifdef HAVE_LIBGCC
281  uint32_t cylinder, temp, head, sector;
282 #endif
283  switch (device->drives[drive].lba_mode) {
284  case ATA_MODE_LBA48:
285  ata_write_port(device, ATA_DRIVE_HEAD_PORT, ATA_DRVSEL_LBA(drive, 0));
286  ata_write_port(device, ATA_SECTOR_COUNT_LOW_PORT, count & 0xFF);
287  ata_write_port(device, ATA_LBA_0_PORT, lba & 0xFF);
288  ata_write_port(device, ATA_LBA_1_PORT, (lba >> 8) & 0xFF);
289  ata_write_port(device, ATA_LBA_2_PORT, (lba >> 16) & 0xFF);
290  ata_write_port(device, ATA_SECTOR_COUNT_HIGH_PORT, (count >> 8) & 0xFF);
291  ata_write_port(device, ATA_LBA_3_PORT, (lba >> 24) & 0xFF);
292  ata_write_port(device, ATA_LBA_4_PORT, (lba >> 32) & 0xFF);
293  ata_write_port(device, ATA_LBA_5_PORT, (lba >> 40) & 0xFF);
294  break;
295  case ATA_MODE_LBA28:
296  ata_write_port(device, ATA_DRIVE_HEAD_PORT, ATA_DRVSEL_LBA(drive, (lba >> 24) & 0xF ));
297  ata_write_port(device, ATA_SECTOR_COUNT_LOW_PORT, count & 0xFF);
298  ata_write_port(device, ATA_LBA_0_PORT, lba & 0xFF);
299  ata_write_port(device, ATA_LBA_1_PORT, (lba >> 8) & 0xFF);
300  ata_write_port(device, ATA_LBA_2_PORT, (lba >> 16) & 0xFF);
301  if (count == 0x100)
302  count = 0;
303  else if (count > 0x100)
304  debugcon_printf("ata: invalid sector count specified for LBA28 drive!");
305  break;
306  case ATA_MODE_CHS:
307 #ifdef HAVE_LIBGCC
308  cylinder = lba / (device->drives[drive].heads * device->drives[drive].sectors_per_track);
309  temp = lba % (device->drives[drive].heads * device->drives[drive].sectors_per_track);
310  head = temp / device->drives[drive].sectors_per_track;
311  sector = temp % device->drives[drive].sectors_per_track + 1;
312  ata_write_port(device, ATA_DRIVE_HEAD_PORT, ATA_DRVSEL_CHS(drive, head & 0xF));
313  ata_write_port(device, ATA_SECTOR_COUNT_LOW_PORT, count & 0xFF);
314  ata_write_port(device, ATA_CYL_LOW_PORT, cylinder & 0xFF);
315  ata_write_port(device, ATA_CYL_HIGH_PORT, (cylinder >> 8) & 0xFF);
316  ata_write_port(device, ATA_SECTOR_NO_PORT, sector & 0xFF);
317 #endif
318  break;
319  }
320 }
