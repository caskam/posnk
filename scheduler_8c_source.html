122 int scheduler_next_task_iterator (llist_t *node, __attribute__((__unused__)) void *param)
123 {
124  scheduler_task_t *task = (scheduler_task_t *) node;
125  if (task == scheduler_current_task)
126  return 0;
127  else if (task->state == PROCESS_READY)
128  return 1;
129  else if (task->state == PROCESS_WAITING) {
130  //earlycon_printf("pid %i is waiting on %x\n",task->pid, task->waiting_on);
131  if (process_was_interrupted(task)) {
132  task->state = PROCESS_INTERRUPTED;
133  task->waiting_on = NULL;
134  return 1;
135  } else if ( (task->wait_timeout_u != 0) && (task->wait_timeout_u <= system_time_micros) ) {
136  task->state = PROCESS_READY;
137  return 1;//TODO: Signal timeout
138  } else if ( (task->wait_timeout_s != 0) && (task->wait_timeout_s <= system_time) ) {
139  task->state = PROCESS_READY;
140  return 1;//TODO: Signal timeout
141  } else if ( (task->waiting_on == NULL) || (!semaphore_try_down(task->waiting_on)) )
142  return 0;
143  task->state = PROCESS_READY;
144  return 1;
145  } else
146  return 0;

150 void schedule()
151 {
152  //TODO: Increment ticks only on preemptive scheduler calls
153  scheduler_current_task->cpu_ticks++;
154  scheduler_task_t *next_task = (scheduler_task_t *) llist_iterate_select(scheduler_task_list, &scheduler_next_task_iterator, NULL);
155  if (scheduler_current_task->state == PROCESS_RUNNING)
156  scheduler_current_task->state = PROCESS_READY;
157  if (next_task != 0) {
158  //earlycon_printf("scheduler switch from %i to %i\n", scheduler_current_task->pid, next_task->pid);
159  scheduler_switch_task(next_task);
160  //earlycon_printf("scheduler switch to %i done, state: %i\n", scheduler_current_task->pid, scheduler_current_task->state);
161  }
162  if ((scheduler_current_task->state != PROCESS_KILLED ) &&(scheduler_current_task->waiting_signal_bitmap != 0))
163  scheduler_switch_task(scheduler_current_task);
164  if (scheduler_current_task->state == PROCESS_READY) {
165  scheduler_current_task->state = PROCESS_RUNNING;
166  return;
167  }
168  if (scheduler_idle_task != 0)
169  scheduler_switch_task(scheduler_idle_task);
170  else
171  return;
172 }
