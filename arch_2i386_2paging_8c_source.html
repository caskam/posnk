160  pd_entry |= table_addr;

162 
163  if (!(flags & PAGING_PAGE_FLAG_USER)) {
164  /* Kernel page! Need to add this table to all directories */
165  pd_entry |= I386_PAGE_FLAG_GLOBAL;

169  } else {

173  }
174  i386_native_flush_tlb_single((uintptr_t)I386_ADDR_TO_PTEPTR(virt_addr));
175  memset(table, 0, sizeof(i386_page_table_t));

178  }

181  /* Tried to map something that was already mapped */
182  /* We will allow this but maybe warn here */

184  table->pages[pt_idx] = 0;
185  }

188  if (flags & PAGING_PAGE_FLAG_USER)
