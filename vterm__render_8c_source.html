31 void vterm_put_char(vterm_t *vterm,chtype c)
32 {
33  static char vt100_acs[]="`afgjklmnopqrstuvwxyz{|}~";
34 
35  if(vterm->ccol >= vterm->cols)
36  {
37  vterm->ccol=0;
38  vterm_scroll_down(vterm);
39  }
40 
41  if(IS_MODE_ACS(vterm))
42  {
43  if(strchr(vt100_acs,(char)c)!=NULL)
44  {
45  vterm->cells[vterm->crow][vterm->ccol].ch=c;//NCURSES_ACS(c);
46  }
47  }
48  else
49  {
50  vterm->cells[vterm->crow][vterm->ccol].ch=c;
51  }
52 
53  vterm->cells[vterm->crow][vterm->ccol].attr=vterm->curattr;
54  vterm->ccol++;
55  vterm_invalidate_cell(vterm, vterm->crow, vterm->ccol - 1);
56 
57  return;
58 }

60 void vterm_render_ctrl_char(vterm_t *vterm,char c)
61 {
62  switch(c)
63  {
64  /* carriage return */
65  case '\r':
66  {
67  vterm->ccol=0;
68  break;
69  }
70 
71  /* line-feed */
72  case '\n':
73  {
74  vterm->ccol=0;
75  vterm_scroll_down(vterm);
76  break;
77  }
78 
79  /* backspace */
80  case '\b':
81  {
82  if(vterm->ccol > 0) vterm->ccol--;
83  break;
84  }
85 
86  /* tab */
87  case '\t':
88  {
89  while(vterm->ccol%8) vterm_put_char(vterm,' ');
90  break;
91  }
92 
93  /* begin escape sequence (aborting previous one if any) */
94  case '\x1B':
95  {
96  vterm_escape_start(vterm);
97  break;
98  }
99 
100  /* enter graphical character mode */
101  case '\x0E':
102  {
103  vterm->state |= STATE_ALT_CHARSET;
104  break;
105  }
106 
107  /* exit graphical character mode */
108  case '\x0F':
109  {
110  vterm->state &= ~STATE_ALT_CHARSET;
111  break;
112  }
113 
114  /* CSI character. Equivalent to ESC [ */
115  case '\x9B':
116  {
117  vterm_escape_start(vterm);
118  vterm->esbuf[vterm->esbuf_len++]='[';
119  break;
120  }
121 
122  /* these interrupt escape sequences */
123  case '\x18':
124  case '\x1A':
125  {
126  vterm_escape_cancel(vterm);
127  break;
128  }
129 
130  /* bell */
131  case '\a':
132  {
133  vterm_handle_bell(vterm);
134  break;
135  }
136 
137 #ifdef DEBUG
138  default:
139  //TODO: fprintf(stderr, "Unrecognized control char: %d (^%c)\n", c, c + '@');
140  break;
141 #endif
142  }
143 }

145 void vterm_render(vterm_t *vterm,const char *data,int len)
146 {
147  int i;
148 
149  for (i = 0; i < len; i++, data++)
150  {
151  /* completely ignore NUL */
152  if(*data == 0) continue;
153 
154  if(*data >= 1 && *data <= 31)
155  {
156  vterm_render_ctrl_char(vterm,*data);
157  continue;
158  }
159 
160  if(IS_MODE_ESCAPED(vterm) && vterm->esbuf_len < ESEQ_BUF_SIZE)
161  {
162  /* append character to ongoing escape sequence */
163  vterm->esbuf[vterm->esbuf_len]=*data;
164  vterm->esbuf[++vterm->esbuf_len]=0;
165 
166  try_interpret_escape_seq(vterm);
167  }
168  else
169  {
170  vterm_put_char(vterm,*data);
171  }
172  }
173  vterm_invalidate_cursor(vterm);
174 }
