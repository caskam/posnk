29 void vterm_interpret_csi(vterm_t *vterm)
30 {
31  static int csiparam[MAX_CSI_ES_PARAMS];
32  int param_count = 0;
33  const char *p;
34  char verb;
35 
36  p=vterm->esbuf+1;
37  verb=vterm->esbuf[vterm->esbuf_len-1];
38 
39  /* parse numeric parameters */
40  while ((*p >= '0' && *p <= '9') || *p == ';' || *p == '?')
41  {
42  if(*p == '?')
43  {
44  p++;
45  continue;
46  }
47 
48  if(*p == ';')
49  {
50  if(param_count >= MAX_CSI_ES_PARAMS) return; /* too long! */
51  csiparam[param_count++]=0;
52  }
53  else
54  {
55  if(param_count==0) csiparam[param_count++] = 0;
56 
57  csiparam[param_count-1] *= 10;
58  csiparam[param_count-1] += *p - '0';
59  }
60 
61  p++;
62  }
63 
64  /* delegate handling depending on command character (verb) */
65  switch (verb)
66  {
67  case 'm':
68  {
69  interpret_csi_SGR(vterm,csiparam,param_count);
70  break;
71  }
72 
73  case 'l':
74  {
75  interpret_dec_RM(vterm,csiparam,param_count);
76  break;
77  }
78 
79  case 'h':
80  {
81  interpret_dec_SM(vterm,csiparam,param_count);
82  break;
83  }
84 
85  case 'J':
86  {
87  interpret_csi_ED(vterm,csiparam,param_count);
88  break;
89  }
90  case 'H':
91  case 'f':
92  {
93  interpret_csi_CUP(vterm,csiparam,param_count);
94  break;
95  }
96  case 'A':
97  case 'B':
98  case 'C':
99  case 'D':
100  case 'E':
101  case 'F':
102  case 'G':
103  case 'e':
104  case 'a':
105  case 'd':
106  case '`':
107  {
108  interpret_csi_CUx(vterm,verb,csiparam,param_count);
109  break;
110  }
111  case 'K':
112  {
113  interpret_csi_EL(vterm,csiparam,param_count);
114  break;
115  }
116  case '@':

118  interpret_csi_ICH(vterm,csiparam,param_count);
119  break;
120  }
121  case 'P':
